git init -- инициализирует репу.

git status -- самая важная команда -- вся информация.

git add file1 file2 -- сказать, чтобы гит следил.

git command --help -- открыть помощь по git command.

git commit -- закоммитить -- зафиксировать изменения.


git log -- показывает состояние последних коммитов.
git log --all --graph

git checkout fb68b8 -- переход в предыдущие коммиты.(после перехода лучше не изменять)

git branch -- вывести все ветки.
git branch newbranch -- создать ветку newbranch.

git merge namebranch -- сливаем namebranch в ту ветку, в которой мы находимся.

git diff -- разница между последней фиксацией файла и последними изменениями.

git push -- залить репу на сервер.
git pull -- докачать репу с сервера.
